Hello Video

This example opens a video capture device using device “0”, reads frames, and shows the video in a GUI window:

package main

import (
	"github.com/hybridgroup/gocv"
)

func main() {
	webcam, _ := gocv.VideoCaptureDevice(0)
	window := gocv.NewWindow("Hello")	
	img := gocv.NewMat()

	for {
		webcam.Read(img)
		window.IMShow(img)
		gocv.WaitKey(1)
	}
}
